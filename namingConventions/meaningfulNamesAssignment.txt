import random


def roll_die(number_of_sides):
    return random.randint(1, number_of_sides)

def main():
    number_of_sides_on_die = 6
    is_rolling = True

    while is_rolling:
        user_response = input("Ready to roll? Enter Q to Quit: ")
        if user_response.lower() != "q":
            rolled_value = roll_die(number_of_sides_on_die)
            print(f"You have rolled a {rolled_value}")
        else:
            is_rolling = False

import random

def is_valid_guess(guess):
    return guess.isdigit() and 1 <= int(guess) <= 100

def main():
    target_number = random.randint(1, 100)
    guessed_correctly = False
    guess = input("Guess a number between 1 and 100: ")
    number_of_guesses = 0

    while not guessed_correctly:
        if not is_valid_guess(guess):
            guess = input("Invalid input. Please enter a number between 1 and 100: ")
            continue
        
        number_of_guesses += 1
        guess = int(guess)

        if guess < target_number:
            guess = input("Too low. Guess again: ")
        elif guess > target_number:
            guess = input("Too high. Guess again: ")
        else:
            print(f"You guessed it in {number_of_guesses} guesses!")
            guessed_correctly = True
    main()

def calculate_armstrong_sum(number):
    total_sum = 0
    digit_count = 0

    # Calculating the number of digits
    temp_number = number
    while temp_number > 0:
        digit_count += 1
        temp_number //= 10

    # Calculating the Armstrong sum
    temp_number = number
    while temp_number > 0:
        digit = temp_number % 10
        total_sum += digit ** digit_count
        temp_number //= 10

    return total_sum

# User Input
user_input = int(input("\nPlease Enter the Number to Check for Armstrong: "))

if user_input == calculate_armstrong_sum(user_input):
    print(f"\n{user_input} is an Armstrong Number.\n")
else:
    print(f"\n{user_input} is Not an Armstrong Number.\n")
